// stores/auth.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { api } from './client'  // ✅ изменено
import { normalizeUser } from './normalizers'  // ✅ изменено
import { useToast } from '../composables/useToast'  // ✅ изменено (не из utils!)
import { useLocalStorage } from '../composables/useLocalStorage'  // ✅ изменено

export const useAuthStore = defineStore('auth', () => {
  const toast = useToast()
  const { data: user, remove: clearUser } = useLocalStorage('currentUser', null)

  const loading = ref(false)
  const error = ref(null)

  const login = async (email, password) => {
    loading.value = true
    error.value = null

    try {
      console.log('🔐 Login attempt:', email)

      // ВРЕМЕННО: используем GET /users/ для демо
      const users = await api.get('/users/')  // ✅ используем api.client
      const foundUser = users.find(u => u.email === email)

      if (!foundUser) {
        throw new Error('Invalid email or password')
      }

      const normalizedUser = normalizeUser(foundUser)
      user.value = normalizedUser

      toast.success(`Welcome back, ${normalizedUser.name}!`)
      return normalizedUser
    } catch (err) {
      console.error('Login error:', err)
      error.value = err.message
      toast.error(err.message)
      throw err
    } finally {
      loading.value = false
    }
  }

  const register = async (email, password, name) => {
    loading.value = true
    error.value = null

    try {
      const newUser = await api.post('/users/', {  // ✅ используем api.client
        email,
        password,
        name,
        role: 'hiring_manager'
      })

      const normalizedUser = normalizeUser(newUser)
      user.value = normalizedUser

      toast.success('Account created successfully!')
      return normalizedUser
    } catch (err) {
      error.value = err.message
      toast.error(err.message)
      throw err
    } finally {
      loading.value = false
    }
  }

  const logout = () => {
    clearUser()
    toast.success('Logged out successfully')
  }

  const isAuthenticated = computed(() => !!user.value)

  return {
    user,
    loading,
    error,
    isAuthenticated,
    login,
    register,
    logout
  }
})
